```typescript
function validate(body: any): BookingData | null {
  const required = ["nombre", "email", "telefono", "servicio", "fecha", "horario", "precio"];
```

```javascript
const getPeriodDates=(filter,from,to)=>{
  const now=new Date();
  const pad=n=>String(n).padStart(2,'0');
  const fmt=d=>`${d.getFullYear()}-${pad(d.getMonth()+1)}-${pad(d.getDate())}`;
  if(filter==="today"){const t=fmt(now);return{from:t,to:t};}
  if(filter==="week"){
    const dow=now.getDay(),diff=dow===0?-6:1-dow;
    const mon=new Date(now);mon.setDate(now.getDate()+diff);
    const sun=new Date(mon);sun.setDate(mon.getDate()+6);
    return{from:fmt(mon),to:fmt(sun)};
  }
  if(filter==="month"){
    const f=new Date(now.getFullYear(),now.getMonth(),1);
    const l=new Date(now.getFullYear(),now.getMonth()+1,0);
    return{from:fmt(f),to:fmt(l)};
  }
  return{from,to};
};
```
